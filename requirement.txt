# Full restore (inside a freshly started Docker container):
#   bash scripts/install_isaac.sh
#
# Or manually:
#   pip install -r ./requirement.txt
#   ISAACLAB_SITE="$(python -m pip show isaaclab | awk -F': ' '/^Location:/ {print $2}')"
#   pip install -e "${ISAACLAB_SITE}/isaaclab/source/isaaclab_assets"
#   pip install -e "${ISAACLAB_SITE}/isaaclab/source/isaaclab_tasks"
#
# The two editable installs are not reachable from this file because they point
# into the `isaaclab` wheel's own `source/` subdirectory after install; the
# helper script resolves the path automatically.
# Use `python -m pip show isaaclab` to locate the installed wheel instead of
# `import isaaclab`: importing `isaaclab` may bootstrap Omniverse Kit and ask
# for the NVIDIA EULA in non-interactive shells.
#
# Verified on:
#   Ubuntu 24.04 (in Docker on WSL2)
#   NVIDIA driver 595.79 on the Windows host
#   Docker run flags: --gpus all -e NVIDIA_DRIVER_CAPABILITIES=all
#                     -e ACCEPT_EULA=Y -e PRIVACY_CONSENT=Y
